1. Overview & Architecture

Connectgold is an edge-computed real-time ad network engineered on Cloudflare Workers and Key-Value (KV) storage. The network uses distributed edge nodes to execute real-time bidding (RTB) auctions, verify publisher placements, enforce frequency capping per client IP, and log hourly and daily impression metrics with near-zero latency.

Network Flow Architecture

[ Publisher Website / App ]
           │
           │ 1. HTTP GET /v1/ad (placement_id, format, consent)
           ▼
[ Cloudflare Edge Worker ] ───► [ Cloudflare KV Store ]
           │                     * Validates Publisher & Consent
           │                     * Queries Active Funded Bids
           │                     * Evaluates Highest eCPM
           │
           │ 2. Returns JSON Payload + Html Snippet
           ▼
[ Publisher Browser / WebView ]
           │
           ├─────────────────────► 3. Renders Ad Unit
           ├─────────────────────► 4. Asynchronous GET /v1/track/impression
           └─────────────────────► 5. On Click: GET /v1/track/click

2. Publisher Integration

Publishers integrate Connectgold by injecting lightweight JavaScript snippets into HTML, or by making server-to-server HTTP API calls.

Standard Banner Units

Banner placements support standard web dimensions (320x50, 300x250, 728x90). The embed snippet handles asynchronous rendering, consent evaluation, and configurable ad rotation.

<!-- Connectgold Ad Network Placement: Leaderboard / Mobile Unit -->
<div id="plc_banner_header" style="width:320px; height:50px; margin:auto;"></div>
<script>
  (function() {
    var placementId = "plc_banner_header";
    var rotationSeconds = 30;

    function fetchAndRenderAd() {
      // Evaluate GDPR consent status from client storage or hardcoded policy
      var consentVal = localStorage.getItem("user_ad_consent") === "granted" ? "1" : "0";
      
      var requestUrl = "https://lingering-sound-9859.trustconnect713.workers.dev/v1/ad" +
        "?placement_id=" + placementId +
        "&format=banner" +
        "&rotation=" + rotationSeconds +
        "&consent=" + consentVal;

      fetch(requestUrl)
        .then(function(res) { return res.json(); })
        .then(function(ad) {
          if (ad.consent_required) return;
          
          var container = document.getElementById(placementId);
          if (container && ad.html_snippet) {
            container.innerHTML = ad.html_snippet;
          } else if (container && ad.payload) {
            container.innerHTML = '<a href="' + ad.click_url + '" target="_blank" rel="noopener">' +
              '<img src="' + ad.payload.image_url + '" width="320" height="50" alt="' + ad.payload.title + '" />' +
              '</a>';
          }

          // Fire impression pixel asynchronously
          if (ad.impression_url) {
            fetch(ad.impression_url);
          }
        })
        .catch(function(err) {
          console.error("[Connectgold] Failed to load banner unit:", err);
        });
    }

    // Initial load
    fetchAndRenderAd();

    // Configure automated client-side rotation
    if (rotationSeconds > 0) {
      setInterval(fetchAndRenderAd, rotationSeconds * 1000);
    }
  })();
</script>

Native In-Feed Units

Native ad units adapt dynamically to publisher container styles. The JSON response exposes raw creative components (title, body, cta_text, image_url) alongside pre-rendered HTML snippets.

<!-- Connectgold Ad Network Placement: Native In-Feed -->
<div id="plc_native_feed"></div>
<script>
  (function() {
    var placementId = "plc_native_feed";
    var requestUrl = "https://lingering-sound-9859.trustconnect713.workers.dev/v1/ad?placement_id=" + placementId + "&format=native&consent=1";

    fetch(requestUrl)
      .then(function(res) { return res.json(); })
      .then(function(ad) {
        var container = document.getElementById(placementId);
        if (!container) return;

        if (ad.html_snippet) {
          container.innerHTML = ad.html_snippet;
        } else {
          container.innerHTML = 
            '<div style="border:1px solid #30363d; padding:16px; background:#161b22; color:#fff; border-radius:8px;">' +
              '<h4 style="margin:0 0 8px 0; color:#fff;">' + ad.payload.title + '</h4>' +
              '<p style="margin:0 0 12px 0; color:#8b949e; font-size:14px;">' + ad.payload.body + '</p>' +
              '<a href="' + ad.click_url + '" target="_blank" style="color:#d4af37; font-weight:bold; text-decoration:none;">' + ad.payload.cta_text + ' &rarr;</a>' +
            '</div>';
        }

        if (ad.impression_url) {
          fetch(ad.impression_url);
        }
      });
  })();
</script>

Rewarded Video Units

Rewarded Video units trigger full-screen or modal overlays and execute server-to-server callback notifications upon completed user views.

function showConnectgoldRewardedAd(userAccountId, onRewardCallback) {
  var placementId = "plc_rewarded_main";
  var callbackUrl = "https://chyke.online/api/reward-callback";
  
  var requestUrl = "https://lingering-sound-9859.trustconnect713.workers.dev/v1/ad" +
    "?placement_id=" + placementId +
    "&format=rewarded" +
    "&consent=1";

  fetch(requestUrl)
    .then(function(res) { return res.json(); })
    .then(function(ad) {
      if (!ad || !ad.payload) {
        alert("No rewarded video campaigns currently available.");
        return;
      }

      console.log("[Connectgold] Streaming video resource:", ad.payload.video_url);

      if (ad.impression_url) fetch(ad.impression_url);

      if (callbackUrl) {
        fetch(callbackUrl + "?user_id=" + encodeURIComponent(userAccountId) + "&status=completed")
          .then(function() {
            if (typeof onRewardCallback === "function") onRewardCallback(true);
          });
      }
    });
}

Interstitial Units

Interstitials launch full-screen ad views during natural application transitions.

function triggerInterstitialAd() {
  var placementId = "plc_interstitial_transition";
  var requestUrl = "https://lingering-sound-9859.trustconnect713.workers.dev/v1/ad?placement_id=" + placementId + "&format=interstitial&consent=1";

  fetch(requestUrl)
    .then(function(res) { return res.json(); })
    .then(function(ad) {
      if (ad && ad.click_url) {
        if (ad.impression_url) fetch(ad.impression_url);
        window.open(ad.click_url, "_blank");
      }
    });
}

3. Privacy, Compliance & GDPR/EEA Consent

Connectgold enforces strict data privacy compliance for traffic originating within European Economic Area (EEA) and UK regions.

Value Behavior
consent=1 User has granted consent. Ad auction runs normally and impression is recorded.
consent=0 User has denied consent or opted out. System blocks non-essential telemetry and serving in EEA regions.
consent=auto Embed script checks local browser storage key user_ad_consent and maps dynamically.

4. Ads.txt Authorization Protocol

To ensure publisher inventory legitimacy and prevent domain spoofing, publishers must host an ads.txt file at their domain root (https://yourdomain.com/ads.txt).

# Connectgold Ad Network Authorized Supply List
lingering-sound-9859.trustconnect713.workers.dev, PUBLISHER_ID, DIRECT, f08c47fec0942fa0
  • Field 1: Network Worker Domain Identifier (lingering-sound-9859.trustconnect713.workers.dev)
  • Field 2: Unique Publisher ID assigned upon registration (e.g., pub_connectgold1234)
  • Field 3: Relationship Type (DIRECT for domain owners, RESELLER for sub-networks)
  • Field 4: Certification Authority ID (f08c47fec0942fa0)

5. Auction Engine & Bid Selection API

The Connectgold auction worker receives HTTP calls, fetches matching ad objects stored under the key prefix ad:FORMAT:*, and ranks them according to maximum eCPM bid rates and remaining wallet budgets.

Endpoint: GET https://lingering-sound-9859.trustconnect713.workers.dev/v1/ad

Request Query Parameters

Parameter Type Required Description
placement_id String Yes Registered publisher placement identifier.
format String Yes Ad unit type (banner, native, interstitial, rewarded).
rotation Integer No Rotation interval in seconds (Default: 0).
consent String Yes Regional consent flag (1 or 0).

6. Impression & Click Measurement Specification

Connectgold uses separate tracking routes to record telemetry asynchronously without blocking main browser rendering threads.

  • Impression Counter Route:
    GET https://floral-pine-5c44.trustconnect713.workers.dev/v1/track/impression?format=banner
  • Click Counter Route:
    GET https://floral-pine-5c44.trustconnect713.workers.dev/v1/track/click?format=banner

Telemetry Storage Mapping in Cloudflare KV

  • Impressions Key: stats:YYYY-MM-DD:FORMAT:impressions
  • Clicks Key: stats:YYYY-MM-DD:FORMAT:clicks
  • Frequency Cap Key: freq:CLIENT_IP:AD_ID

7. Third-Party Analytics Hooks

Publishers can automatically forward Connectgold ad engagement events to Google Analytics 4, Plausible, or Mixpanel by configuring client keys in the network dashboard.

Google Analytics 4 (GA4)

if (typeof gtag === 'function') {
  gtag('event', 'connectgold_ad_impression', {
    'ad_id': ad.ad_id,
    'placement_id': placementId,
    'format': 'banner'
  });
}

Plausible Analytics

if (typeof plausible === 'function') {
  plausible('Connectgold Impression', {
    props: { ad_id: ad.ad_id, placement_id: placementId }
  });
}

Mixpanel

if (typeof mixpanel === 'object' && typeof mixpanel.track === 'function') {
  mixpanel.track('Connectgold Impression', {
    'ad_id': ad.ad_id,
    'placement_id': placementId
  });
}
